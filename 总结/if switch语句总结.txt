//switch

swtich(变量)
{
case(变量数值);
  命令;
  break;
case(变量数值)    //共同执行下行命令 
case(变量数值);
  命令;
  break;
case(变量数值);
  命令;
  break;
default:
  命令;
  break;
}
 注意：switch变量中只能为int型
